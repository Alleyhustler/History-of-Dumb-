	Tutorial de Pascal parte 06



	»»»»»»»»»»»»»»» Manipulando Arquivos «««««««««««««««


1. Declaraçao.
   Veja o exemplo:

Type Arq = File Of Integer;
Var  Arquivo : Arq;

	Aqui a variavel 'Arquivo' corresponde a arquivos de numeros inteiros.
Os arquivos nos programas em pascal podem ser:

Integer		=	File de Numeros Inteiros
Real		=	File de Numeros reais
Records		=	File de dados

	No caso de arquivo de dados, devem aparecer assim:

Type Pessoa = Record
     Nome   : String[30];
     Idade  : Integer;
     Sexo   : Char;
     Altura : Real;
End;
Var Arquivo : File Of Pessoa;



2. Nomeando Arquivos
   Comando : Assign
   Sintaxe : Assign(varievel,nome_do.arq);
   Exemplo :

Type Arq = File Of Integer;
Var  Arquivo : Arq;
Begin
     Assign(Arquivo,'A:EXEMPLO.TXT');
(* Com isso nomeamos a Variavel Arquivo com o A:EXEMPLO.TXT *)
End.



3. Criando arquivos
   Comando : Rewrite.
   Sintaxe : Rewrite(variavel);
   Exemplo :

Type Arq = File Of Integer;
Var  Arquivo : Arq;
Begin
     Assign(Arquivo,'A:EXEMPLO.TXT');
     Rewrite(Arquivo);
(* Com isso criamos o Arquivo EXEMPLO.TXT na unidade A: *)
End.



4. Abrindo arquivos
   Comando : Reset
   Sintaxe : Reset(variavel);
   Exemplo :

Type Arq = File Of Integer;
